    "Content-Type" = "application/json"
    "accept"       = "application/json"
}

# 3. Định nghĩa cấu hình Index (Serverless cho tiết kiệm)

    name      = "cloud-sentinel-index"
    dimension = 1536
    metric    = "cosine"
    spec      = @{
        serverless = @{

            cloud  = "aws"

            region = "us-east-1"
        }
    }
} | ConvertTo-Json

# 4. Gọi API để tạo
Invoke-RestMethod -Uri "https://api.pinecone.io/indexes" -Method Post -Headers $headers -Body $body
